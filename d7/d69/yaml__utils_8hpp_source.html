111 {
112 const auto mapNode = n[key];
113 if (!mapNode || !mapNode.IsMap())
114 return;
115
116 using Scalar = typename Vec::Scalar;
117
118 for (auto it : mapNode)
119 {
120 int start = it.first.as<int>(); // key: 0, 8, 10, ...
121
122 const YAML::Node &val = it.second;
123
124 if (val.IsSequence())
125 {
126 // Sequence: write multiple entries starting at 'start'.
127 for (std::size_t k = 0; k < val.size(); ++k)
128 {
129 int idx = start + static_cast<int>(k);
130 if (idx >= 0 && idx < v.size())
131 v(idx) = val[k].as<Scalar>();
132 }
133 }
134 else
135 {
136 // Scalar: write a single element at 'start'.
137 int idx = start;
138 if (idx >= 0 && idx < v.size())

259 for (int r = 0; r < L; ++r)
